Q1. LIST ALL THE EMPLOYEES WHOSE SALARY IS NULL 
              SELECT * FROM EMPLOYE WHERE SALARY IS NULL;

Q2. LIST ALL THE EMPLOYEES WHO DON’T HAVE A DEPT NO
              SELECT * FROM EMPLOYE  WHERE DEPTNO IS NULL;

Q3. LIST ALL THE SALESMEN IN DEPT 30
              SELECT * FROM EMPLOYE  WHERE DESIGNATION='SALESMAN' AND DEPTNO=30;

Q4. LIST ALL THE SALESMEN IN DEPT NUMBER 30 AND HAVING SALARY GREATER THAN 1500
              SELECT * FROM EMPLOYE  WHERE DESIGNATION='SALESMAN' AND DEPTNO=30 AND SALARY>1500;

Q5. LIST ALL THE EMPLOYEES WHOSE NAME STARTS WITH ‘S’ OR ‘A’
              SELECT * FROM EMPLOYE  WHERE ENAME LIKE 'S%' OR ENAME LIKE 'A%';

Q6. LIST ALL THE EMPLOYEES EXCEPT THOSE WHO ARE WORKING IN DEPT 10 & 20. 
              SELECT * FROM EMPLOYE WHERE DEPTNO NOT IN(10,20);

Q7. LIST THE EMPLOYEES WHOSE NAME DOES NOT START WITH ‘S’
              SELECT * FROM EMPLOYE WHERE ENAME NOT LIKE 'S%';

Q8. LIST ALL THE EMPLOYEES WHO ARE HAVING REPORTING MANAGERS IN DEPT 10
              SELECT * FROM EMPLOYE  WHERE MANAGER IS NOT NULL AND DEPTNO=10;

Q9. LIST ALL THE EMPLOYEES WHOSE SALARY IS NULL AND WORKING AS CLERK
              SELECT * FROM EMPLOYE WHERE SALARY IS NULL AND DESIGNATION='CLERK';

Q10. LIST ALL THE EMPLOYEES WHO DON’T HAVE A REPORTING MANAGER IN DEPTNO 10 OR 30
               SELECT *  FROM EMPLOYE WHERE MANAGER IS NULL AND DEPTNO IN(10,30);

Q11. LIST ALL THE SALESMEN IN DEPT 30 WITH SAL MORE THAN 2450

Q12.LIST ALL THE ANALYST IN DEPT NUMBER 20 AND HAVING SALARY GREATER THAN 2500

Q13.LIST ALL THE EMPLOYEES WHOSE NAME STARTS WITH ‘M’ OR ‘J’

Q14.LIST ALL THE EMPLOYEES WITH ANNUAL SALARY EXCEPT THOSE WHO ARE WORKING IN DEPT 30 
               SELECT ENAME, SALARY*12 ANNUAL_SALARY FROM EMPLOYE WHERE DEPTNO NOT IN (30);

Q15.LIST THE EMPLOYEES WHOSE NAME DOES NOT END WITH ‘ES’ OR ‘R’

Q16.LIST ALL THE EMPLOYEES WHO ARE HAVING REPORTING MANAGERS IN DEPT 10 ALONG WITH 10% HIKE IN SALARY

Q19.DISPLAY ALL THE EMPLOYEE WHO ARE JOINED IN FEB

Q20.LIST THE EMPLOYEES WHO ARE NOT WORKING AS MANAGERS AND CLERKS IN DEPT 10 AND 20 WITH A SALARY IN THE RANGE OF 1000 TO 3000
                 SELECT * FROM EMPLOYE WHERE DESIGNATION NOT IN('MANAGER','CLERK') AND DEPTNO IN(10,20) AND SALARY BETWEEN 1000 AND 3000;

Q21.LIST THE EMPLOYEES WHOSE SALARY NOT IN THE RANGE OF 1000 TO 2000 AND WORKING IN DEPT 10,20 OR 30 EXCEPT ALL SALESMEN
